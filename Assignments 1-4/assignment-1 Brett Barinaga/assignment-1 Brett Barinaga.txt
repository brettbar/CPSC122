#include <iostream> 
#include <Windows.h>
#include <fstream>

using namespace std;

int main()
{	
	int sum = 0; 
	ofstream file;
	file.open("Numbers.txt");
	// At this point I have created a blank file called Numbers.
	for (int i = 1; i <= 100; i++)
	{
		file << i << endl;
	}
	file.close();
	// At this point i have printed all the numbers from 1 to 100 to the Numbers file.
	ifstream inFile;
	int value;
	
	inFile.open("Numbers.txt");
	// At this point I have opened the Numbers file as an in File, and in the following for loop I will go through and display each value, while also keeping a running sum.
	while (inFile >> value)
	{
		cout << value << endl;

		sum = sum + value;
	}

	inFile.close();

	cout << "The total sum is " << sum << endl;
	return 0;
}
